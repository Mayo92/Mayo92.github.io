interface Simple {

SimpleClass与SimpleClass2

//没有open不能被继承
open class SimpleClass(var x:Int,val y:String): AbsClass(),Simple {

    val z:Long = 0L

    //需要override
    override fun absMethod() {}

    //默认有get/set，只读的话没有set
    override val simpleProperty: Int
        get() {
            return 2
        }

    override fun simpleMethod() {}

    fun y(){}

    fun zzz(string:String){}

    //前面加上final不让子类继承
    override fun overridable() {
    }
}

class SimpleClass2(x:Int,y:String):SimpleClass(x,y){

    override fun overridable() {
    }
}

1.6 属性

property = field+ getter + setter

例子

class Person(age:Int,name:String) {

    var age:Int = age
    get() {
        return field//backing field
    }
    set(value) {
        println("setAge:$value")
        field = value
    }

    var name:String = name
}


fun main() {
    //属性引用
    val ageRef = Person::age
    val person = Person(10,"yly")
    val nameRef = person::name
    nameRef.set("krido")
}

class PoorGuy{
    var pocket:Double = 0.0
}

//扩展方法
fun PoorGuy.noMoney(){}

//扩展成员
//属性=backing field（状态，存储空间）+getter（方法，接口行为）+setter（方法，接口行为）
//backing field是类的状态，这个值与类密切绑定
//扩展成员里扩展属性是没有办法定义backing field
//扩展成员是没有办法存储状态的，因为没有口袋不能存，但是可以访问类里面做存储操作，且只能访问公共的
var PoorGuy.moneyLeft:Double
    get() {
//        return 0.0
        //类里加上pocket可以这么写，pocket不能是private
        return this.pocket
    }
    set(value) {
        //类里加上pocket可以这么写，pocket不能是private
        pocket = value
    }

interface Guy{
    //接口里的属性没有backing field，接口没有状态不能存储东西只能定义行为
    //这里定义get/set是默认行为
    var moneyLeft:Double
    get() {
        return 0.0
    }
    set(value) {

    }

    fun noMoney(){
        //定义默认行为
    }
}
